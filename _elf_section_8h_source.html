214 enum class EElfSectionFlags : UWord
215 {
216 WRITE = (1 << 0), /* Writable */
217 ALLOC = (1 << 1), /* Occupies memory during execution */
218 EXECINSTR = (1 << 2), /* Executable */
219 MERGE = (1 << 4), /* Might be merged */
220 STRINGS = (1 << 5), /* Contains nul-terminated strings */
221 INFO_LINK = (1 << 6), /* `sh_info' contains SHT index */
222 LINK_ORDER = (1 << 7), /* Preserve order after combining */
223 OS_NONCONFORMING = (1 << 8), /* Non-standard OS specific handling required */
224 GROUP = (1 << 9), /* Section is member of a group. */
225 TLS = (1 << 10), /* Section hold thread-local data. */
226 COMPRESSED = (1 << 11), /* Section with compressed data. */
227 MASKOS = 0x0ff00000, /* OS-specific. */
228 MASKPROC = 0xf0000000, /* Processor-specific */
229 GNU_RETAIN = (1 << 21), /* Not to be GCed by linker. */
230 ORDERED = (1 << 30), /* Special ordering requirement (Solaris). */
231 EXCLUDE = (1U << 31) /* Section is excluded unless referenced or allocated (Solaris).*/
232 };
