5/* How to extract and insert information held in the symbol Info field. */
6
7#define ELF32_ST_BIND(val) (((unsigned char)(val)) >> 4)
8#define ELF32_ST_TYPE(val) ((val)&0xf)
9#define ELF32_ST_INFO(bind, type) (((bind) << 4) + ((type)&0xf))
10
11/* Both ElfSymbol32 and ElfSymbol64 use the same one-byte symbol Info field. */
12#define ELF64_ST_BIND(val) ELF32_ST_BIND(val)
13#define ELF64_ST_TYPE(val) ELF32_ST_TYPE(val)
14#define ELF64_ST_INFO(bind, type) ELF32_ST_INFO((bind), (type))

16/* How to extract and insert information held in the symbol Other field. */
